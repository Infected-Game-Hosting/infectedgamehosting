name: Deploy to IONOS VPS

on:
  push:
    branches:
      - main # Change to 'production' or whatever branch you want to auto-deploy
  workflow_dispatch: # Allows manual trigger from GitHub UI

jobs:
  deploy:
    name: Build and Deploy
    runs-on: ubuntu-latest

    steps:
      - name: üõéÔ∏è Checkout Code
        uses: actions/checkout@v4

      - name: üîß Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: "18" # Or your Node version
          cache: "npm"

      - name: üì¶ Install Dependencies
        run: npm ci

      - name: üèóÔ∏è Build React App
        run: npm run build
        env:
          CI: false # Prevents warnings from failing the build
          # Add any environment variables your build needs
          # REACT_APP_API_URL: ${{ secrets.API_URL }}

      - name: üì§ Deploy to VPS
        uses: appleboy/scp-action@v0.1.7
        with:
          host: ${{ secrets.VPS_HOST }}
          username: ${{ secrets.VPS_USERNAME }}
          key: ${{ secrets.VPS_SSH_KEY }}
          port: ${{ secrets.VPS_PORT }}
          source: "dist/*" # ‚Üê Changed from build/* to dist/*
          target: "/home/github-deploy/build"
          strip_components: 1
          rm: true

      - name: üöÄ Execute Deployment Script
        uses: appleboy/ssh-action@v1.0.3
        with:
          host: ${{ secrets.VPS_HOST }}
          username: ${{ secrets.VPS_USERNAME }}
          key: ${{ secrets.VPS_SSH_KEY }}
          port: ${{ secrets.VPS_PORT }}
          script: |
            cd ~
            ./deploy.sh

      - name: ‚úÖ Deployment Complete
        run: |
          echo "üéâ Deployment successful!"
          echo "üåê Site: https://infectedgamehosting.com"
